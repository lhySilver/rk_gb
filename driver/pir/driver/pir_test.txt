static int fpir = -1;
#define PIR_MAGIC         'p'
#define PIR_SET_GPIO           _IOW(PIR_MAGIC, 0xa0, struct PIR_GPIO )
#define PIR_SET_RUN            _IOW(PIR_MAGIC, 0xa1, int )  //0-stop 1-start
#define PIR_SET_SEN            _IOW(PIR_MAGIC, 0xa2, int )  //0-低 1-中 2-高

struct PIR_GPIO {
    int gpio_out;
    int gpio_sen;
};

int SystemPirDet(void)
{
	if (fpir < 0)
	{
		fpir = open("/dev/pir", O_RDWR | O_NONBLOCK);
		if(fpir < 0)
		{
			perror("open error!\n");
			return -1;
		}
	}

	static int pro = 0;
	if(pro == 0)
	{
		pro = 1;

		int run = 1;
		if(ioctl(fpir, PIR_SET_RUN, &run) < 0)
		{
			printf("set pir run error!\n");
			return -1;
		}
		else
		{
			printf("set pir run  ok\n");
		}
	}
	
	int ret;
    struct pollfd fds;//定义一个pollfd结构体key_fds
    unsigned int val = 0;
    memset(&fds,0,sizeof(fds));

    fds.fd = fpir;//文件
    fds.events = POLLIN;//poll直接返回需要的条件

	ret = poll((struct pollfd *)&fds, 1, 200);//调用sys_poll系统调用，如果200毫秒内没有产生POLLIN事件，那么返回，如果有POLLIN事件，直接返回
	if(!ret)
	{
		// printf("get pir out time out\n");
		return -1;
	}
	else
	{
		if(fds.revents == POLLIN)//如果返回的值是POLLIN，说明有数据POLL才返回的
		{
			ret = read(fpir, &val, sizeof(val));//读取pir输出值
			if (ret <=0)
			{
				printf("read pir out value error\n");
				return -1;
			}
			
			// printf("get status:0x%x\n",val);
			return val;
		}
	}
    
	return -1;
}

int SystemSetPirSen(int val,int option)//设置pir灵敏度
{
	if (fpir < 0)
	{
		fpir = open("/dev/pir", O_RDWR | O_NONBLOCK);
		if(fpir < 0)
		{
			perror("open error!\n");
			return -1;
		}
	}
	static int s_v = -1;
	int value = val;
	if (s_v == val)
	{
		return 0;
	}
	s_v = val;

	//设置灵敏度
	if(ioctl(fpir, PIR_SET_SEN, &value) < 0)
	{
		printf("set pir sen error!\n");
		return -1;
	}
	else
	{
		printf("set pir sen  ok\n");
		return 0;
	}
	
	return 0;
}

static int g_pir_det_result_M = 0; //中间
static int g_pir_det_result_R = 0; //右
static int g_pir_det_result_L = 0; //左

static int g_pir_run = 1;

void milliseconds_sleep(unsigned long mSec)
{
    struct timeval tv;
    tv.tv_sec=mSec/1000;
    tv.tv_usec=(mSec%1000)*1000;
    int err;
    do
	{
		err=select(0,NULL,NULL,NULL,&tv);
    }
	while(err<0 && errno==EINTR);
}

void *thread_pir_proc(void *arg)
{
	prctl(PR_SET_NAME, "thread_pir_proc", 0, 0, 0);

	int value = 0;
	while(g_pir_run)
	{
		//1中，2右，3左
		value = SystemPirDet();
		// AppErr("SystemPirDet=%d\n",value);
		if (value == -1)
		{
			g_pir_det_result_M = 0;
			g_pir_det_result_R = 0;
			g_pir_det_result_L = 0;
		}
		else
		{
			if (value & 0x01)//pir 1
			{
				g_pir_det_result_M = 1;
			}
			if ( value & 0x02)//pir 2
			{
				g_pir_det_result_R = 1;
			}
			if ( value& 0x04) //pir 3
			{
				g_pir_det_result_L = 1;
			}
		}

		milliseconds_sleep(5);
	}

	return NULL;
}
